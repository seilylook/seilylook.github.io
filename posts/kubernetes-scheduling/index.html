---
apiVersion: apps/v1
kind: ReplicaSet
metadata:
   name: replicaset-1
spec:
   replicas: 2
   selector:
      matchLabels:
        tier: front-end
   template:
     metadata:
       labels:
        tier: nginx(X) -> front-end(O)
     spec:
       containers:
       - name: nginx
         image: nginx

Status: Running

controlplane ~ ➜  vim bee-pod.yaml

apiVersion: v1
kind: Pod
metadata:
  name: bee
spec:
  containers:
    - image: nginx
      name: bee
  tolerations:
    - key: spray
      value: mortein
      effect: NoSchedule
      operator: Equal
:wq

controlplane ~ ➜  kubectl create -f bee-pod.yaml
